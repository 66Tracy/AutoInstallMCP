<project_specification>
  <project_name>MCP Auto-Installer — Agent-Driven Dockerfile Builder (Demo)</project_name>

  <overview>
    一个精简的多Agent系统：给定本地MCP Server仓库路径，自动分析其结构与运行时依赖，
    生成Dockerfile，构建镜像并测试MCP协议握手，最终输出可直接运行的容器配置。

    三个Agent通过Orchestrator串联：
      RepoAnalysisAgent → DockerfileAgent → BuildTestAgent
    其中DockerfileAgent与BuildTestAgent形成"生成→构建→报错→修复"的循环，直到成功或达到重试上限。
  </overview>

  <requirement>
    使用uv来构建环境。
  </requirement>

  <technology_stack>
    <language>Python 3.11+</language>
    <llm_sdk>OpenAI Python SDK（兼容任意OpenAI-compatible endpoint）</llm_sdk>
    <containerization>Docker SDK for Python (docker-py)</containerization>
    <config>Pydantic v2 用于数据校验与schema定义</config>
    <cli>简单的argparse入口（demo阶段不引入Click/Typer）</cli>
    <logging>Python标准logging</logging>
  </technology_stack>

    <design_principles>
      - 三个Agent，职责清晰，不过度拆分。
      - Agent之间不直接调用，Orchestrator统一调度。
      - DockerfileAgent与BuildTestAgent通过Orchestrator形成修复循环。
      - 第一版仅支持本地仓库路径输入，预留url扩展接口。
      - 所有Agent输入输出均为Pydantic model，保证类型安全。

  <core_data_schemas>
    <description>
      集中定义在 src/schemas/models.py。每个Agent产出一个Result model，
      Orchestrator负责传递。
    </description>

    <models>
      class EnvVarSpec(BaseModel):
          name: str                              # 如 "OPENAI_API_KEY"
          description: str                       # 用途说明
          required: bool
          default_value: str | None = None
          source: str                            # 发现来源："README" / ".env.example" / "source_code"
          category: str                          # "api_key" / "database" / "oauth" / "config" / "other"

      class RepoAnalysisResult(BaseModel):
          """RepoAnalysisAgent的输出，合并了仓库结构分析与Secrets分析"""
          # —— 仓库结构 ——
          repo_path: str                         # 本地仓库路径
          language: str                          # python / typescript / go / rust
          package_manager: str                   # pip / npm / yarn / pnpm / cargo / go mod
          entry_point: str                       # 启动MCP server的命令或文件
          install_commands: list[str]            # 按序的安装命令
          config_files_found: list[str]          # 找到的配置文件列表
          readme_summary: str                    # README中安装说明的摘要
          base_image_suggestion: str             # 建议的Docker基础镜像
          transport_type: str                    # "stdio" / "sse" / "streamable-http" / "unknown"
          exposed_port: int | None = None        # SSE/HTTP时需要暴露的端口
          extra_system_deps: list[str]           # 需要apt-get安装的系统依赖
          # —— 运行时依赖/Secrets ——
          required_env_vars: list[EnvVarSpec]    # 必须提供的环境变量
          optional_env_vars: list[EnvVarSpec]    # 可选的环境变量
          env_file_template: str                 # 生成的 .env.example 内容
          secrets_risk: str                      # "low" / "medium" / "high"
          # —— 元信息 ——
          confidence: float                      # 0.0-1.0 自评置信度
          notes: list[str]                       # 观察、警告

      class DockerfileSpec(BaseModel):
          """DockerfileAgent的输出"""
          dockerfile_content: str                # 完整Dockerfile文本
          env_vars_to_inject: list[str]          # 运行时需注入的ENV名称
          expected_transport: str
          exposed_ports: list[int]
          entrypoint: list[str]                  # ENTRYPOINT指令
          notes: list[str]

      class BuildResult(BaseModel):
          """构建阶段结果"""
          success: bool
          image_tag: str
          build_log: str                         # 完整构建日志
          error_summary: str | None = None       # 失败时的错误摘要

      class TestResult(BaseModel):
          """测试阶段结果"""
          container_started: bool
          startup_log: str                       # 容器启动日志（前100行）
          mcp_handshake_ok: bool                 # MCP initialize握手是否成功
          error_summary: str | None = None
          missing_secrets: list[str]             # 因缺少secrets导致的运行失败

      class FinalOutput(BaseModel):
          """Pipeline最终输出"""
          success: bool
          repo_path: str
          dockerfile_content: str
          env_template: str                      # .env.example内容
          image_tag: str
          transport_type: str
          required_secrets: list[EnvVarSpec]
          mcp_handshake_tested: bool
          startup_command: str                   # 可直接运行的docker run命令
          notes: list[str]
    </models>
  </core_data_schemas>

  <agent_specifications>

    <agent name="RepoAnalysisAgent">
      <purpose>
        分析本地MCP仓库的项目结构、语言、依赖、安装方式、MCP传输类型，
        同时扫描运行时所需的环境变量与Secrets。合并输出为RepoAnalysisResult。
      </purpose>
      <tools>
        - list_directory_tree(path) → str
        - read_file(path, max_lines=500) → str
        - search_files(root, glob_pattern) → list[str]
        - search_in_files(root, regex, file_pattern="*") → list[dict]
      </tools>
      <workflow>
        1. list_directory_tree 了解项目布局。
        2. 读取关键文件：README.md, package.json / pyproject.toml / Cargo.toml / go.mod。
        3. 识别语言、包管理器、安装命令、入口点。
        4. 在源码中搜索 "stdio" / "sse" / "StreamableHTTP" 等关键词判断传输类型。
        5. 搜索 .env / .env.example / .env.sample 文件。
        6. 在源码中搜索 os.environ / process.env / env::var 等模式。
        7. 搜索常见secrets模式：*_API_KEY, *_SECRET, *_TOKEN, DATABASE_URL。
        8. 交叉对比README文档中的配置说明。
        9. 分类每个变量：required/optional，api_key/database/oauth/config。
        10. 生成 .env.example 模板。
        11. 输出 RepoAnalysisResult。
      </workflow>
      <max_rounds>15</max_rounds>
    </agent>

    <agent name="DockerfileAgent">
      <purpose>
        根据RepoAnalysisResult生成Dockerfile。同时承担"修复者"角色：
        当收到BuildTestAgent的错误反馈时，分析报错并生成修正后的Dockerfile。
        生成与修复共享同一Agent实例，通过不同的user message触发。
      </purpose>
      <input>
        首次调用：RepoAnalysisResult
        修复调用：RepoAnalysisResult + BuildResult/TestResult（含错误信息）
      </input>
      <tools>
        - read_file(path) → str                  # 查阅仓库文件辅助修复
        - write_file(path, content) → str        # 写出Dockerfile
      </tools>
      <workflow_generate>
        1. 根据语言和版本选择base image（slim/alpine优先）。
        2. 编译型语言使用multi-stage build。
        3. 先COPY依赖清单文件（利用Docker缓存层）。
        4. 安装系统依赖 → 安装项目依赖 → COPY源码。
        5. 声明ENV变量占位（带注释说明）。
        6. 配置EXPOSE（如SSE/HTTP）。
        7. 设置ENTRYPOINT/CMD。
        8. 使用非root用户运行。
        9. 写出Dockerfile到磁盘。
        10. 输出DockerfileSpec。
      </workflow_generate>
      <workflow_fix>
        1. 读取错误摘要和完整build/test日志。
        2. 分类错误：依赖安装失败 / 缺系统库 / base image版本不对 /
           入口点错误 / 权限问题 / 端口问题。
        3. 必要时读取仓库文件获取线索（CI配置、Makefile等）。
        4. 生成修正后的Dockerfile。
        5. 输出更新的DockerfileSpec。
      </workflow_fix>
      <max_rounds>10</max_rounds>
    </agent>

    <agent name="BuildTestAgent">
      <purpose>
        构建Docker镜像，启动容器，执行MCP协议握手测试。
        输出BuildResult + TestResult供Orchestrator判断是否需要触发修复循环。
      </purpose>
      <input>DockerfileSpec + RepoAnalysisResult</input>
      <tools>
        - build_image(context_path, dockerfile_path, tag, timeout=600) → BuildResult
        - run_container(image_tag, env_vars, ports, timeout=30) → str (container_id)
        - get_container_logs(container_id, tail=100) → str
        - send_mcp_initialize(target, transport) → str
        - stop_and_remove_container(container_id) → bool
      </tools>
      <workflow>
        1. 构建镜像，捕获完整build日志。
        2. 构建失败 → 返回BuildResult(success=False)，不进入测试。
        3. 构建成功 → 启动容器（required secrets用空值占位）。
        4. 等待容器就绪，抓取启动日志。
        5. stdio传输：通过docker exec向stdin管道发送MCP initialize消息。
           SSE/HTTP传输：向endpoint发送HTTP POST。
        6. 记录握手结果。
        7. 分析日志中因缺少secrets导致的错误。
        8. 清理容器。
        9. 输出BuildResult + TestResult。
      </workflow>
      <max_rounds>8</max_rounds>
    </agent>

  </agent_specifications>

  <orchestrator_specification>
    <pipeline>
      INIT → REPO_ANALYSIS → DOCKERFILE_GEN → BUILD_TEST → (FIX_LOOP) → DONE / FAILED

      修复循环（FIX_LOOP）：
        BUILD_TEST失败 → DockerfileAgent.fix() → BUILD_TEST → ...
        最多循环 max_fix_attempts 次（默认3次）。
        如果失败仅因secrets缺失 → 视为DONE（附warnings）。
    </pipeline>

    <state>
      class PipelineState(Enum):
          INIT = "init"
          REPO_ANALYSIS = "repo_analysis"
          DOCKERFILE_GEN = "dockerfile_gen"
          BUILD_TEST = "build_test"
          FIXING = "fixing"
          DONE = "done"
          FAILED = "failed"

      class PipelineContext(BaseModel):
          state: PipelineState = PipelineState.INIT
          repo_path: str
          work_dir: str
          repo_analysis: RepoAnalysisResult | None = None
          dockerfile_spec: DockerfileSpec | None = None
          build_result: BuildResult | None = None
          test_result: TestResult | None = None
          fix_attempts: int = 0
          max_fix_attempts: int = 3
          error_history: list[str] = []
    </state>

    <orchestrator_logic>
      def run(repo_path: str) -> FinalOutput:
          ctx = PipelineContext(repo_path=repo_path, work_dir=tempdir)

          # Stage 1: 仓库分析（结构 + Secrets）
          ctx.repo_analysis = repo_analysis_agent.run(repo_path)
          if ctx.repo_analysis.confidence < 0.3:
              return FinalOutput(success=False, notes=["仓库分析置信度过低"])

          # Stage 2: 生成Dockerfile
          ctx.dockerfile_spec = dockerfile_agent.generate(ctx.repo_analysis)

          # Stage 3+: 构建测试循环
          while ctx.fix_attempts <= ctx.max_fix_attempts:
              build_result, test_result = build_test_agent.run(ctx.dockerfile_spec, ctx.repo_analysis)

              if build_result.success and test_result.mcp_handshake_ok:
                  return assemble_final_output(ctx, success=True)

              if not build_result.success or is_code_error(test_result):
                  ctx.fix_attempts += 1
                  ctx.error_history.append(build_result.error_summary or test_result.error_summary)
                  ctx.dockerfile_spec = dockerfile_agent.fix(ctx.repo_analysis, build_result, test_result)
                  continue

              if is_secrets_only_error(test_result):
                  return assemble_final_output(ctx, success=True, warnings=["需提供secrets才能完整运行"])

          return FinalOutput(success=False, notes=["达到最大修复次数仍未成功"])
    </orchestrator_logic>
  </orchestrator_specification>

  <base_agent_specification>
    <interface>
      class BaseAgent(ABC):
          agent_id: str
          msg_thread: MessageThread
          max_round: int
          tool_registry: dict[str, Callable]

          + run(input_data: Any) → Any               # 主入口
          + add_system_message(msg: str)
          + add_user_message(msg: str)
          + execute_tool(call: FunctionCallIntent) → str
          + get_tool_definitions() → list[dict]      # OpenAI tool schema
          + should_stop() → bool                     # 轮次上限检查

          @abstractmethod
          + init_msg_thread() → None
          @abstractmethod
          + get_available_tools() → list[str]
    </interface>
  </base_agent_specification>

      - stdio: docker exec管道JSON-RPC到进程stdin，读取stdout。
      - SSE/HTTP: POST到 http://localhost:{port}/sse 或 /mcp。
      - 未知传输类型：先试stdio，再试HTTP。
